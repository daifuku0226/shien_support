    <script>
        function showToast(title, desc) {
            const toast = document.getElementById('download-toast');
            document.getElementById('toast-title').innerText = title;
            document.getElementById('toast-desc').innerText = desc;
            toast.classList.remove('hidden');
            setTimeout(() => {
                toast.classList.add('hidden');
            }, 4000);
        }

        const BASE_URL = "https://daifuku0226.github.io/shien_support/";
        const FILE_URLS = {
            guide_pdf: BASE_URL + "Special_Needs_Planning_Guide.pdf",
            support_files: [
                BASE_URL + "01.docx",
                BASE_URL + "02.docx",
                BASE_URL + "03.docx",
                BASE_URL + "04.docx",
                BASE_URL + "05.docx"
            ],
            instruction_docx: BASE_URL + "05.docx"
        };

        /**
         * PDFを別タブで表示（これは動作が安定しています）
         */
        function openGuidePdf() {
            window.open(FILE_URLS.guide_pdf, '_blank');
        }

        /**
         * ファイルをダウンロードさせる補助関数
         */
        function triggerDownload(url, filename) {
            const a = document.createElement('a');
            a.href = url;
            a.download = filename || url.split('/').pop();
            // セキュリティ対策としてターゲットを_blankに設定する場合もあるが
            // ダウンロードの場合はこのままDOMに追加してクリック
            document.body.appendChild(a);
            a.click();
            setTimeout(() => document.body.removeChild(a), 100);
        }

        /**
         * 5つのファイルを一括ダウンロード
         */
        function downloadSupportFiles() {
            showToast("一括ダウンロード", "5つのファイルを取得します。ブラウザの許可を確認してください。");
            
            FILE_URLS.support_files.forEach((url, index) => {
                // ブラウザのブロックを回避するため、実行間隔を空ける
                setTimeout(() => {
                    triggerDownload(url);
                }, index * 800); 
            });
        }

        /**
         * 指導計画（05.docx）をダウンロード
         */
        function downloadInstructionPlan() {
            showToast("ダウンロード開始", "指導計画様式(05.docx)を保存します。");
            triggerDownload(FILE_URLS.instruction_docx, "05.docx");
        }
    </script>
